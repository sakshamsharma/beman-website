28// This is a minimal contiguous iterator. It uses stl_interfaces library from Boost
29// (current implementation based on https://wg21.link/P2727R4).
30//
31// TODO: Change this to use the stl_interfaces library from Beman if/when available.
32//
33// @tparam T - The type of the elements the iterator points to.
34// @tparam Container - The type of the container the iterator points to. This parameter exists solely so that different
35// containers using this template can instantiate different types, even if the T parameter is the same.
36template <class T, class Container>
37struct contiguous_iterator : public base_contiguous_iterator<T, Container> {
38 // Alias for the base class.
39 using base_type = base_contiguous_iterator<T, Container>;
40 // Alias for types from the base class.
41 using typename base_type::difference_type;
42 using typename base_type::pointer;
43 using typename base_type::reference;
44
45 // Default constructor.
46 constexpr contiguous_iterator() noexcept : m_current() {}
47
48 // Pointer to iterator constructor.
49 constexpr contiguous_iterator(pointer it) noexcept : m_current(it) {}
50
51 // As per P2727R4, for contiguous iterator we only need to provide operator*, operator+= and operator-.
52 constexpr reference operator*() const noexcept { return *m_current; }
53 constexpr auto& operator+=(difference_type pos) noexcept {
54 m_current += pos;
55 return *this;
56 }
57 constexpr difference_type operator-(contiguous_iterator other) const noexcept {
58 return m_current - other.m_current;
59 }
60
61 private:
62 T* m_current;
63};
